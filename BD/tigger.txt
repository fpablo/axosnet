CREATE TRIGGER [dbo].[Trigger_Date] ON [dbo].[Doctos] FOR INSERT AS BEGIN

Declare @Id int
set @Id = (select DoctoID from inserted)

Update [dbo].[Doctos]
Set Fecha = GetDate(),FechaHoraCreacion=  GetDate(), FechaHoraModificacion=GetDate()
Where DoctoID = @Id
END
